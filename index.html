What's functional programming?

What are functions?

  • Computation is function application.
  • Functions consist of operations and any number of free variables to be used in them.
  • Application requires resolving those variables.

public static <T> T reduce(final T[] input, final Function<T, T> mapper,
        final Function<T, Function<T, T>> reducer, final T initial) {
        
        final Collection<T> mapped =
                Collections2.transform(Arrays.asList(input), mapper);
        T result = initial;
        for(T value : mapped){
            final Function<T, T> toApply = reducer.apply(result);
            result = toApply.apply(value);
        }
        return result;
}

package java.util.function;
@FunctionalInterface
public interface Function<T, R> {
    R apply(T t);  
    default <V> Function<V, R>
        compose(Function<? super V, ? extends T> before) {
        Objects.requireNonNull(before);
        return (V v) -> apply(before.apply(v));
    }
    default <V> Function<T, V>
        andThen(Function<? super R, ? extends V> after) {
        Objects.requireNonNull(after);
        return (T t) -> after.apply(apply(t));
    }
    static <T> Function<T, T> identity() {
        return t -> t;
    }
}
